22   public class Audio {
23       private static Audio instance = new Audio();
24       private final AudioPlayerManager playerManager;
25       private final Map<Long, GuildMusicManager> musicManagers;
26       private Audio(){
27           this.musicManagers = new HashMap<>();
28           this.playerManager = new DefaultAudioPlayerManager();
29           AudioSourceManagers.registerRemoteSources(playerManager);
30           AudioSourceManagers.registerLocalSource(playerManager);
31       }
32   
33       private synchronized GuildMusicManager getGuildAudioPlayer(Guild guild){
34           long guildId  =guild.getIdLong();
35           GuildMusicManager musicManager = musicManagers.get(guildId);
36           if(musicManager == null){
37               musicManager = new GuildMusicManager(playerManager);
38               musicManagers.put(guildId, musicManager);
39           }
40           guild.getAudioManager().setSendingHandler(musicManager.getSendHandler());
41           return musicManager;
42       }
43       public void loadAndPlay(final TextChannel channel, final String trackUrl){
44           RestAction<Message> msg = channel.getMessageById(channel.getLatestMessageId());
45           GuildMusicManager musicManager = getGuildAudioPlayer(channel.getGuild());
46           playerManager.loadItemOrdered(musicManager, trackUrl, new AudioLoadResultHandler() {
47               @Override
48               public void trackLoaded(AudioTrack track) {
49                   channel.sendMessage("Ajout de "+track.getInfo().title+" à la file d'attente").queue(message -> {
50                       message.addReaction(Main.getClient().getSuccess());
51                   });
52                   play(channel.getGuild(), musicManager, track);
53               }
54   
55               @Override
56               public void playlistLoaded(AudioPlaylist playlist) {
57                   AudioTrack firstTrack = playlist.getSelectedTrack();
58                   if(firstTrack == null){
59                       firstTrack = playlist.getTracks().get(0);
60                   }
61                   channel.sendMessage("Ajout de "+firstTrack.getInfo().title+" à la file d'attente.").queue();
62                   msg.queue(message -> {
63                       message.addReaction(Main.getClient().getSuccess());
64                   });
65                   play(channel.getGuild(), musicManager, firstTrack);
66               }
67   
68               @Override
69               public void noMatches() {
70                   msg.queue(message -> {
71                       message.addReaction(Main.getClient().getWarning());
72                   });
73                   channel.sendMessage("Aucun morceau n'a été trouvé à l'adresse "+trackUrl).queue();
74               }
75   
76               @Override
77               public void loadFailed(FriendlyException e) {
78                   msg.queue(message -> {
79                       message.addReaction(Main.getClient().getError());
80                       channel.sendMessage("Impossible de jouer le morceau demandé: "+e.getMessage()).queue();
81                   });
82               }
83           });
84       }
85   
86       private void play(Guild guild, GuildMusicManager musicManager, AudioTrack track) {
87           connectToFirstVoiceChannel(guild.getAudioManager());
88           musicManager.scheduler.queue(track);
89       }
90       public String setVolume(Guild guild, int v){
91           GuildMusicManager musicManager = getGuildAudioPlayer(guild);
92           musicManager.player.setVolume(v);
93           musicManager.player.setPaused(false);
94           return "New volume :" +musicManager.player.getVolume();
95       }
96       public void skipTrack(TextChannel channel){
97           GuildMusicManager musicManager = getGuildAudioPlayer(channel.getGuild());
98           musicManager.scheduler.nextTrack();
99           channel.sendMessage("Saut du morceau actuel").queue();
100      }
101      public void switchPause(Guild guild){
102          GuildMusicManager musicManager = getGuildAudioPlayer(guild);
103          if(musicManager.player.isPaused()){
104              musicManager.player.setPaused(false);
105          }else{
106              musicManager.player.setPaused(true);
107          }
108      }
109      private static void connectToFirstVoiceChannel(AudioManager audioManager){
110          if(!audioManager.isConnected() && !audioManager.isAttemptingToConnect()){
111              for(VoiceChannel voiceChannel : audioManager.getGuild().getVoiceChannels()){
112                  audioManager.openAudioConnection(voiceChannel);
113                  break;
114              }
115          }
116      }
117  
118      public static Audio getInstance() {
119          return new Audio();
120      }
121  }
