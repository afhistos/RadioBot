1    package be.afhistos.discord.commands;
2    
3    import com.jagrosh.jdautilities.command.Command;
4    import com.jagrosh.jdautilities.command.CommandEvent;
5    import com.jagrosh.jdautilities.commons.JDAUtilitiesInfo;
6    import com.jagrosh.jdautilities.doc.standard.CommandInfo;
7    import com.jagrosh.jdautilities.examples.doc.Author;
8    import javafx.application.Application;
9    import net.dv8tion.jda.bot.entities.ApplicationInfo;
10   import net.dv8tion.jda.core.EmbedBuilder;
11   import net.dv8tion.jda.core.JDAInfo;
12   import net.dv8tion.jda.core.Permission;
13   import org.slf4j.Logger;
14   import org.slf4j.LoggerFactory;
15   
16   import java.awt.*;
17   
18   @CommandInfo(name = "About", description = "Donne quelques infos à propos de moi ;)")
19   @Author("afhistos")
20   public class CommandAbout extends Command {
21       private boolean is_author = false;
22       private final Color color;
23       private String replacement_icon = "+";
24       private String description;
25       private final Permission[] permissions;
26       private String oauthLink;
27       private final String[] features;
28   
29       public CommandAbout(Color color, String description, String[] features, Permission... permissions){
30           this.color = color;
31           this.description=description;
32           this.features=features;
33           this.name="about";
34           this.help="Donne quelques infos à propos de moi ;)";
35           this.guildOnly=false;
36           this.permissions=permissions;
37           this.botPermissions= new Permission[] {Permission.MESSAGE_EMBED_LINKS};
38       }
39       @Override
40       protected void execute(CommandEvent event) {
41           if(oauthLink == null){
42               try{
43                   ApplicationInfo info = event.getJDA().asBot().getApplicationInfo().complete();
44                   oauthLink = info.isBotPublic() ? info.getInviteUrl(0L, permissions) : "";
45               }catch (Exception e){
46                   Logger log = LoggerFactory.getLogger("OAuth2");
47                   log.error("Impossible de générer le lien d'invitation");
48               }
49           }
50           EmbedBuilder builder = new EmbedBuilder();
51           builder.setColor(event.getGuild() == null ? color : event.getGuild().getSelfMember().getColor());
52           builder.setAuthor("Tout à propos de "+event.getSelfUser().getName() +"!", null, event.getSelfUser().getAvatarUrl());
53           boolean join = !(event.getClient().getServerInvite() == null || event.getClient().getServerInvite().isEmpty());
54           boolean inv = !oauthLink.isEmpty();
55           String invLine = "\n"+(join?"Rejoint mon serveur [`ici`]("+event.getClient().getServerInvite()+")" : (inv ? "Tu peux m'" : ""))+(inv?(join? ", ou ":"")+"[`inviter`]("+oauthLink+") sur ton serveur":"")+"!";
56           String author = event.getJDA().getUserById(event.getClient().getOwnerId())==null ? "<@"+event.getClient().getOwnerId()+">":event.getJDA().getUserById(event.getClient().getOwnerId()).getName();
57           StringBuilder descr = new StringBuilder().append("Salut! Je suis **").append(event.getSelfUser().getName()).append("**, ")
58                   .append(description).append("\nJ'").append(is_author ? "'ai été écrit en Java par" : "appartient à").append(" **")
59                   .append(author).append("** en utilisant [Les extensions de commandes](" + JDAUtilitiesInfo.GITHUB + ") (" + JDAUtilitiesInfo.AUTHOR + " ")
60                   .append(JDAUtilitiesInfo.VERSION).append(") et la  [Librairie JDA](https://github.com/DV8FromTheWorld/JDA) (")
61                   .append(JDAInfo.VERSION).append(")\nÉcris `").append(event.getClient().getTextualPrefix()).append(event.getClient().getHelpWord())
62                   .append("` afin de voir toutes mes commandes!").append(join || inv ? invLine : "").append("\n\nVoici un résumé de ce que je peux faire: ```css");
63           for (String feature : features)
64               descr.append("\n").append(event.getClient().getSuccess().startsWith("<") ? replacement_icon : event.getClient().getSuccess()).append(" ").append(feature);
65           descr.append(" ```");
66           builder.setDescription(descr);
67           if (event.getJDA().getShardInfo() == null)
68           {
69               builder.addField("Stats", event.getJDA().getGuilds().size() + " Serveurs \n1 Shard", true);
70               builder.addField("Utilisateurs", event.getJDA().getUsers().size() + " Unique\n" + event.getJDA().getGuilds().stream().mapToInt(g -> g.getMembers().size()).sum() + " Total", true);
71               builder.addField("Salons", event.getJDA().getTextChannels().size() + " Textuels\n" + event.getJDA().getVoiceChannels().size() + " Vocaux", true);
72           }
73           else
74           {
75               builder.addField("Stats", (event.getClient()).getTotalGuilds() + " Serveurs\nShard " + (event.getJDA().getShardInfo().getShardId() + 1)
76                       + "/" + event.getJDA().getShardInfo().getShardTotal(), true);
77               builder.addField("Ce \"shard\"", event.getJDA().getUsers().size() + " Utilisateurs\n" + event.getJDA().getGuilds().size() + " Serveurs", true);
78               builder.addField("", event.getJDA().getTextChannels().size() + " Salons textuels\n" + event.getJDA().getVoiceChannels().size() + " Salons vocaux", true);
79           }
80           builder.setFooter("Dernier redémarrage", null);
81           builder.setTimestamp(event.getClient().getStartTime());
82           event.reply(builder.build());
83       }
84   }
