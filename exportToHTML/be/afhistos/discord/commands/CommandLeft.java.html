1    package be.afhistos.discord.commands;
2    
3    import com.jagrosh.jdautilities.command.Command;
4    import com.jagrosh.jdautilities.command.CommandEvent;
5    import com.jagrosh.jdautilities.doc.standard.CommandInfo;
6    import com.jagrosh.jdautilities.examples.doc.Author;
7    import net.dv8tion.jda.core.managers.AudioManager;
8    
9    @CommandInfo(name = "exit", description = "Permet de me déconnecter du salon vocal")
10   @Author("afhistos")
11   public class CommandLeft extends Command {
12       public CommandLeft(){
13           this.name = "exit";
14           this.help = "Permet de couper la radio";
15           this.guildOnly = true;
16           this.category = new Category("Gestion de la radio");
17           this.requiredRole = "DJ";
18       }
19       @Override
20       protected void execute(CommandEvent event) {
21           AudioManager audioManager = event.getGuild().getAudioManager();
22           if(event.getSelfMember().getVoiceState().inVoiceChannel()){
23               audioManager.closeAudioConnection();
24               event.getMessage().clearReactions().complete();
25               try {
26                   wait();
27               } catch (InterruptedException e) {
28                   e.printStackTrace();
29               }
30               if(event.getSelfMember().getVoiceState().inVoiceChannel()){
31                   event.getMessage().clearReactions().complete();
32                   event.reactError();
33                   event.reply("Une erreur est apparue (Like a pokemon)! Réessaie.");
34               }else {
35                   event.reactSuccess();
36               }
37           } else {
38               event.getMessage().clearReactions().complete();
39               event.reactWarning();
40               event.reply("Il faut que je sois dans un salon vocal pour le quitter chef !");
41           }
42       }
43   }
